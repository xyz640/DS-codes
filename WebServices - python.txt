sample1.py
from flask import Flask, jsonify, request
# Fixed: Flask initialization requires __name__ as an argument
app = Flask(__name__)
@app.route('/add', methods=['POST'])
def add():
    # Fixed: All lines below must be indented to be part of the function
    data = request.get_json()
    a = data['a']
    b = data['b']
    result = a + b
    answer = jsonify({'result': result})
    return answer

if __name__ == '__main__':
    app.run()
